def self.erb(html_file)
  file = html_file.to_s + '.erb'
  ERB.new(File.read(file)).result(binding)
end

def self.redirect(path)
  hash = {}
  hash[:status] = 302
  hash[:body] = ''
  hash[:headers] = { 'Location' => path }
  hash
end
